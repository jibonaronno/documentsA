uint16_t CRC16cal(char *buffer, uint16_t size)
{
	uint16_t i=0;
	uint16_t j=0;
	uint16_t CRC = 0xFFFF;
	uint8_t  CRCHi = 0xFF;
	uint8_t  CRCLo = 0xFF;
	uint8_t  CRCLSB = 0x00;
	
	for (i=0; i < size; i++)
	{
		CRC = CRC ^ buffer[i];
		
		for (j=0; j < 8; j++)
		{
			CRCLSB = (CRC & 0x0001);
			CRC = ((CRC >> 1) & 0x7FFF);
			
			if (CRCLSB)
			{
				CRC = (CRC ^ 0xA001);
			}
		}
	}
	
	return CRC;
	
}
